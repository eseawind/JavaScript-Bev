Examples

var  myMenuPanel = new Bev.MenuPanel({
    "body":$("#div")
});

 myMenuPanel.addMenuItem(
      {
          "icon":"imagepath",
          "hover_icon":"imagepath"
          "title":"基本操作",
          "menu":new Bev.Menu()//$("<div>").html("this is a div")
      }
 );
